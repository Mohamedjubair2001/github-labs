pipeline {

    agent any

    tools {
        maven 'mymaven'
    }

    stages {

        stage('Step 1 : Clone Repository') {
            steps {
                echo "Cloning repository from GitHub"
                git 'https://github.com/akshu20791/addressbook-cicd-project'
            }
        }

        stage('Step 2 : Compile Code') {
            steps {
                echo "Compiling the project"
                sh 'mvn clean compile'
            }
        }

        stage('Step 3 : Code Review (PMD)') {
            steps {
                echo "Running PMD code analysis"
                sh 'mvn pmd:pmd'
            }
            post {
                success {
                    recordIssues tools: [pmdParser(pattern: '**/pmd.xml')]
                }
            }
        }

        stage('Step 4 : Unit Testing') {
            steps {
                echo "Running unit tests"
                sh 'mvn test'
            }
            post {
                success {
                    junit 'target/surefire-reports/*.xml'
                }
            }
        }

        stage('Step 5 : SonarQube Analysis') {
            steps {
                echo "Starting SonarQube scan"
                withSonarQubeEnv('sonarqubeserver') {
                    sh 'mvn clean verify sonar:sonar'
                }
            }
        }

        stage('Step 6 : Package Application') {
            steps {
                echo "Packaging the application"
                sh 'mvn package'
            }
        }

        stage('Deployment of the App') {
            steps {
                sh "sudo mv ${WORKSPACE}/target/addressbook.war /home/ubuntu/apache-tomcat-9.0.115/webapps/"
            }
        }
        stage('Quality Gate') {
            steps {
                timeout(time: 10, unit: 'MINUTES') {
                    waitForQualityGate abortPipeline: true
                }
            }
        }
    }
}
